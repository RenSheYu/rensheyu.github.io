A. 回文

【题解】

PAM裸题. 考虑回文串的一个回文子串，分两类讨论，一类包含最后一个字符即母串后缀，跳回文树即可. 另一类删去首尾两个字符，沿 PAM dfs 一遍即可.

跳回文树时记得打上标记，使子树内节点不再访问，同时保证时间复杂度. 效率 O(∑n)O(\sum n)O(∑n). 期望得分：100.

【代码】

#include<bits/stdc++.h>
const int maxn=5000000+10;
char s[maxn];bool vis[maxn];
int T,n,len[maxn],fail[maxn],ch[maxn][26],cnt,last;
long long Ans,ans[maxn];
inline void dfs ( int u,int fr )
{
	int dep=0,x=u;vis[x]=true;
	while ( fail[x]>1 and !vis[fail[x]] ) x=fail[x],vis[x]=true,dep++;
	ans[u]=dep+((fr>1)?(ans[fr]+1):0);Ans+=ans[u];
	for ( int i=0;i<26;i++ ) if ( ch[u][i] ) dfs(ch[u][i],u);
	x=u;vis[x]=false;
	while ( dep-- ) x=fail[x],vis[x]=false;
}
int main()
{
	for ( scanf("%d",&T);T--; )
	{
		scanf(" %s",s+1);n=strlen(s+1);Ans=0;
		last=cnt=0;s[0]='#';fail[0]=1;len[++cnt]=-1;
		for ( int i=1;i<=n;i++ )
		{
			while ( s[i-len[last]-1]!=s[i] ) last=fail[last];
			if ( !ch[last][s[i]-97] )
			{
				len[++cnt]=len[last]+2;
				int j=fail[last];
				while ( s[i-len[j]-1]!=s[i] ) j=fail[j];
				fail[cnt]=ch[j][s[i]-97];ch[last][s[i]-97]=cnt;
			}
			last=ch[last][s[i]-97];
		}
		dfs(0,0);dfs(1,1);printf("%lld\n",Ans);
		for ( int i=0;i<=cnt;i++ )
		{
			for ( int j=0;j<26;j++ ) ch[i][j]=0;
			fail[i]=ans[i]=len[i]=0;
		}
	}
	return 0;
}

B. Sum of Prefix Sums

【题解】

显然考虑点分治. 考虑两条链的答案如何合并.

记 sumsumsum 为前缀和，SSS 为前缀和的和，需要合并的链为 a,ba,ba,b，其中 aaa 长度 xxx，bbb 长度 yyy. 则有答案为 Sa+y×suma+SbS_a+y \times sum_a +S_bSa​+y×suma​+Sb​.

显然 SbS_bSb​ 可以单独处理，且 aaa 一定时 Sa+y×sumaS_a+y \times sum_aSa​+y×suma​ 是关于 yyy 的一次函数. 李超线段树维护即可.

【代码】

#include<bits/stdc++.h>
const int maxn=150000+10;
struct tree { long long a,b; } t[maxn<<2];
struct node { long long val1,val2,sum,l,x; } st[maxn];
int siz[maxn],son[maxn],tp,n,root,dep[maxn],m;
long long a[maxn],ans;bool vis[maxn];
std::vector<int> e[maxn];
#define ls (k<<1)
#define rs (k<<1|1)
#define mid ((l+r)>>1)
inline void build ( int k,int l,int r )
{
	t[k].a=t[k].b=0;
	if ( l==r ) return;
	build(ls,l,mid);build(rs,mid+1,r);
}
inline void update ( int k,int l,int r,long long a,long long b )
{
	if ( t[k].a*l+t[k].b<=a*l+b and t[k].a*r+t[k].b<=a*r+b ) { t[k].a=a;t[k].b=b;return; }
	if ( t[k].a*l+t[k].b>=a*l+b and t[k].a*r+t[k].b>=a*r+b ) return;
	if ( t[k].a*mid+t[k].b<a*mid+b ) std::swap(t[k].a,a),std::swap(t[k].b,b);
	if ( a<t[k].a ) update(ls,l,mid,a,b);
	else update(rs,mid+1,r,a,b);
}
inline long long query ( int k,int l,int r,int x )
{
	if ( l==r ) return t[k].a*x+t[k].b;
	if ( x<=mid ) return std::max(t[k].a*x+t[k].b,query(ls,l,mid,x));
	else return std::max(t[k].a*x+t[k].b,query(rs,mid+1,r,x));
}
#undef ls
#undef rs
#undef mid
inline void getroot ( int u,int fr,int size )
{
	siz[u]=1;son[u]=0;
	for ( int v:e[u] ) if ( !vis[v] and v!=fr ) getroot(v,u,size),siz[u]+=siz[v],son[u]=std::max(son[u],siz[v]);
	son[u]=std::max(son[u],size-son[u]);
	if ( son[u]<son[root] ) root=u;
}
inline void dfs ( int u,int fr,long long val1,long long val2,long long sum,int rt )
{
	dep[u]=dep[fr]+1;m=std::max(m,dep[u]);siz[u]=1;
	for ( int v:e[u] ) if ( !vis[v] and v!=fr ) dfs(v,u,val1+sum+a[v],val2+a[v]*dep[u],sum+a[v],rt),siz[u]+=siz[v];
	if ( siz[u]==1 ) st[++tp]=(node){val1,val2,sum,dep[u],rt};
}
inline void solve ( int u )
{
	vis[u]=true;tp=0;m=dep[u]=1;
	for ( int v:e[u] ) if ( !vis[v] ) dfs(v,u,2*a[u]+a[v],a[v]*dep[u],a[u]+a[v],v);
	st[++tp]=(node){a[u],0,a[u],1,0};st[0].x=st[tp+1].x=-1;
	for ( int i=1;i<=tp;i++ ) st[i].sum-=a[u];
	build(1,1,m);
	for ( int i=1,j;i<=tp;i=j )
	{
		for ( j=i;st[j].x==st[i].x;j++ ) ans=std::max(ans,query(1,1,m,st[j].l)+st[j].val1);
		for ( j=i;st[j].x==st[i].x;j++ ) update(1,1,m,st[j].sum,st[j].val2);
	}
	build(1,1,m);
	for ( int i=tp,j;i>=1;i=j )
	{
		for ( j=i;st[j].x==st[i].x;j-- ) ans=std::max(ans,query(1,1,m,st[j].l)+st[j].val1);
		for ( j=i;st[j].x==st[i].x;j-- ) update(1,1,m,st[j].sum,st[j].val2);
	}
	for ( int v:e[u] ) if ( !vis[v] ) root=0,getroot(v,u,siz[v]),solve(root);
}
signed main()
{
	scanf("%d", &n);
	for ( int i=1,u,v;i<n;i++ ) scanf("%d%d",&u,&v),e[u].push_back(v),e[v].push_back(u);
	for ( int i=1;i<=n;i++ ) scanf("%lld",&a[i]);
	son[0]=n;getroot(1,0,n);solve(root);
	return !printf("%lld\n",ans);
}

C. Tritwise Mex

【题解】

设 c=a∗bc=a*bc=a∗b，记 aia_{i}ai​ 表示数列 aaa 所有最低位为 iii 的项依序组成的数列，显然有 ：

c0=a1∗b1+a1∗b2+a2∗b1+a2∗b2=(a1+a2)∗(b1+b2)c_{0}=a_{1}*b_{1}+a_{1}*b_{2}+a_{2}*b_{1}+a_{2}*b_{2}=(a_{1}+a_{2})*(b_{1}+b_{2}) c0​=a1​∗b1​+a1​∗b2​+a2​∗b1​+a2​∗b2​=(a1​+a2​)∗(b1​+b2​)

c1=a0∗b0+a0∗b2+a2∗b0=(a0+a1+a2)∗(b0+b1+b2)−(a1+a2)∗(b1+b2)−a0∗b1−a1∗b0c_{1}=a_{0}*b_{0}+a_{0}*b_{2}+a_{2}*b_{0}=(a_{0}+a_{1}+a_{2})*(b_{0}+b_{1}+b_{2})-(a_{1}+a_{2})*(b_{1}+b_{2})-a_{0}*b_{1}-a_{1}*b_{0} c1​=a0​∗b0​+a0​∗b2​+a2​∗b0​=(a0​+a1​+a2​)∗(b0​+b1​+b2​)−(a1​+a2​)∗(b1​+b2​)−a0​∗b1​−a1​∗b0​

c2=a0∗b1+a1∗b0c_{2}=a_{0}*b_{1}+a_{1}*b_{0} c2​=a0​∗b1​+a1​∗b0​

于是可以用 444 次乘法完成一维计算，效率 O(4k−3k)O(4^k - 3^k)O(4k−3k).

【代码】

#include<bits/stdc++.h>
inline std::vector<long long> mul ( const std::vector<int> &A,const std::vector<int> &B,int n )
{
	if ( n==1 )
	{
		std::vector<long long> C;
		C.push_back(1LL*A[0]*B[0]);
		return C;
	}
	n/=3;
	std::vector<int> A0,B0;
	for ( int i=0;i<n;i++ ) A0.push_back(A[i+n*2]),B0.push_back(B[i+n*2]);
	std::vector<long long> C1=mul(A0,B0,n);
	for ( int i=0;i<n;i++ ) A0[i]+=A[i],B0[i]+=B[i];
	std::vector<long long> C2=mul(A0,B0,n);
	for ( int i=0;i<n;i++ ) A0[i]+=A[i+n],B0[i]+=B[i+n];
	std::vector<long long> C3=mul(A0,B0,n);
	for ( int i=0;i<n;i++ ) A0[i]-=A[i],B0[i]-=B[i];
	std::vector<long long> C=mul(A0,B0,n);
	for ( int i=0;i<n;i++ ) C.push_back(C2[i]-C1[i]);
	for ( int i=0;i<n;i++ ) C.push_back(C3[i]-C[i]-C[i+n]);
	return C;
}
inline int read ( void )
{
	int x=0;char ch;
	while ( !isdigit(ch=getchar()) ) ;
	for ( x=ch^48;isdigit(ch=getchar()); ) x=(x<<1)+(x<<3)+(ch^48);
	return x;
}
inline void write ( long long x,char ch )
{
	int st[50]={0},tp=0;
	while ( x ) st[++tp]=x%10,x/=10;
	if ( !tp ) st[++tp]=0;
	for ( int i=tp;i;i-- ) putchar(st[i]^48);
	putchar(ch);
}
signed main()
{
	int n=read(),m=1;
	for ( int i=1;i<=n;i++ ) m*=3;
	std::vector<int> A,B;
	for ( int i=0;i<m;i++ ) A.push_back(read());
	for ( int i=0;i<m;i++ ) B.push_back(read());
	std::vector<long long> C=mul(A,B,m);
	for ( int i=0;i<m;i++ ) write(C[i]," \n"[i==m-1]);
	return 0;
}
