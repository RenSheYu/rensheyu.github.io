A. 迷惑数字统计

【题解】

真的很迷惑.

考虑和为 10n10^n10n 的要求，显然需要将所有数字两两配对，配对完需要保证有恰好一对和为 101010，剩下的对和为 000 或 999.

考虑枚举和为 101010 的一对，剩下的用背包转移. 显然 111 和 888，222 和 777，333 和 666，444 和 555 都是唯一对应的，直接转移. 而对于 000 和 999，只需要保证 000 比 999 多的个数为偶数即可，单独转移.

效率 O(m2)O(m^2)O(m2). 其中 mmm 为 ? 个数.

【代码】

#include<bits/stdc++.h>
const int mod=1000000007;
int n,m,cnt[20],tot[20],siz[20],f[20][2000],fct[200000],inv[200000],ans;
char s[200000];
inline int power ( int x,int y )
{
	int z=1;
	for ( ;y;y>>=1,x=1LL*x*x%mod ) if ( y&1 ) z=1LL*z*x%mod;
	return z;
}
inline int C ( int n,int m ) { return 1LL*fct[n]*inv[m]%mod*inv[n-m]%mod; }
inline void solve ( int x,int y )
{
	for ( int i=0;i<=9;i++ ) siz[i]=0,cnt[i]=tot[i];
	if ( !cnt[x] ) siz[x]++;
	else cnt[x]--;
	if ( !cnt[y] ) siz[y]++;
	else cnt[y]--;
	memset(f,0,sizeof(f));
	for ( int i=0;i<=m;i++ ) for ( int j=0;i+j<=m;j++ )
	{
		int A=i+cnt[0],B=j+cnt[9];
		if ( i<siz[0] or j<siz[9] ) continue;
		if ( siz[9] ) B--;
		if ( B>A or ((A-B)&1) ) continue;
		f[1][i+j]=(f[1][i+j]+1LL*C(m,i)*C(m-i,j))%mod;
	}
	for ( int i=1;i<=4;i++ )
	{
		int j=9-i,A=siz[i],B=siz[j];
		if ( cnt[i]<cnt[j] ) A+=cnt[j]-cnt[i];
		else B+=cnt[i]-cnt[j];
		for ( int p=0;A+B+2*p<=m;p++ ) for ( int k=0;k+A+B+2*p<=m;k++ ) if ( f[i][k] )
			f[i+1][k+A+B+2*p]=(f[i+1][k+A+B+2*p]+1LL*C(m-k,A+p)*C(m-k-A-p,B+p)%mod*f[i][k])%mod;
	}
	ans=(ans+f[5][m])%mod;
}
signed main()
{
	scanf(" %s",s+1);n=strlen(s+1);fct[0]=inv[0]=inv[1]=1;
	for ( int i=1;i<=n;i++ ) fct[i]=1LL*fct[i-1]*i%mod;
	for ( int i=2;i<=n;i++ ) inv[i]=1LL*(mod-mod/i)*inv[mod%i]%mod;
	for ( int i=3;i<=n;i++ ) inv[i]=1LL*inv[i-1]*inv[i]%mod;
	for ( int i=1;i<=n;i++ )
		if ( s[i]=='?' ) m++;
		else tot[s[i]^48]++;
	for ( int i=1;i<=5;i++ ) solve(i,10-i);
	return !printf("%d\n",ans);
}

B. 校运会

【题解】

根据题意，所求即为一个二分图最大权匹配. 直接建图跑最大费用可行流即可.

比较可以直接用数组实现.

注意：由于不要求得奖人数最多，应跑最大费用而不要求满流.

【代码】

#include<bits/stdc++.h>
const int inf=1<<29;
int n,m,k;
struct Cost
{
	int cnt[12];
	Cost(int x=0,int y=0)
	{
		memset(cnt,0,sizeof(cnt));
		if ( x ) cnt[x]+=y;
	}
	inline friend Cost operator + ( const Cost &u,const Cost &v )
	{
		Cost sum;
		for ( int i=1;i<=k;i++ ) sum.cnt[i]=u.cnt[i]+v.cnt[i];
		return sum; 
	}
	inline friend Cost operator * ( const Cost &u,const int &v )
	{
		Cost sum;
		for ( int i=1;i<=k;i++ ) sum.cnt[i]=u.cnt[i]*v;
		return sum; 
	}
};
inline int cmp ( const Cost &u,const Cost &v )
{
	for ( int i=1;i<=k;i++ )
		if ( u.cnt[i]>v.cnt[i] ) return 1;
		else if ( u.cnt[i]<v.cnt[i] ) return -1;
	return 0;
}
struct Flow
{
	int n,S,T,cnt,h[200000],fr[200000];
	Cost dis[200000];bool vis[200000];
	struct edge { int v,nxt,f;Cost c; } e[200000];
	inline void addedge ( int u,int v,int w )
	{
		e[++cnt].nxt=h[u];e[h[u]=cnt].v=v;e[cnt].f=1;e[cnt].c=Cost(w,1);
		e[++cnt].nxt=h[v];e[h[v]=cnt].v=u;e[cnt].f=0;e[cnt].c=Cost(w,-1);
	}
	inline void init ( int n ) { S=++n;T=++n;this->n=n;cnt=1; }
	inline bool spfa ( void )
	{
		for ( int i=1;i<=n;i++ ) vis[i]=false,dis[i]=Cost(1,-inf),fr[i]=0;
		std::queue<int> q;q.push(S);dis[S]=Cost();vis[S]=true;
		while ( !q.empty() )
		{
			int u=q.front();q.pop();vis[u]=false;
			for ( int i=h[u];i;i=e[i].nxt ) if ( e[i].f and cmp(dis[u]+e[i].c,dis[e[i].v])>0 )
			{
				dis[e[i].v]=dis[u]+e[i].c;fr[e[i].v]=i;
				if ( !vis[e[i].v] ) vis[e[i].v]=true,q.push(e[i].v);
			}
		}
		return cmp(dis[T],Cost())>0;
	}
	inline Cost maxcost_maxflow ( void )
	{
		Cost ans;
		while ( spfa() )
		{
			int res=inf;
			for ( int i=T;i!=S;i=e[fr[i]^1].v ) res=std::min(res,e[fr[i]].f);
			ans=ans+dis[T]*res;
			for ( int i=T;i!=S;i=e[fr[i]^1].v ) e[fr[i]].f-=res,e[fr[i]^1].f+=res;
		}
		return ans;
	}
} G;
inline int read ( void )
{
	int x=0;char ch;
	while ( !isdigit(ch=getchar()) ) ;
	for ( x=ch^48;isdigit(ch=getchar()); ) x=(x<<1)+(x<<3)+(ch^48);
	return x;
}
signed main()
{
	n=read();m=read();k=read();G.init(n+m);
	for ( int i=1;i<=n;i++ ) for ( int t=read(),x,y;t--; ) x=read(),y=read(),G.addedge(i,x+n,y);
	for ( int i=1;i<=n;i++ ) G.addedge(G.S,i,0);
	for ( int i=1;i<=m;i++ ) G.addedge(i+n,G.T,0);
	Cost Ans=G.maxcost_maxflow();
	for ( int i=1;i<=k;i++ ) printf("%d%c",Ans.cnt[i]," \n"[i==k]);
	return 0;
}

C. 寻找字符串

【题解】

考虑对询问建 AC 自动机.

显然所要求的即为折线形的字符串. 考虑将折线两端延长至边界处，则询问串为新得到的串的子串. 同时这样得到的串只有 R×CR \times CR×C 个. 因此枚举每个串在 failfailfail 树上跑一遍，对于每个询问串查询 failfailfail 树所在节点子树内的访问次数和即可. 注意细节与去重. 效率 O(n3+∑∣s∣)O(n^3+\sum \lvert s \rvert)O(n3+∑∣s∣).

【代码】

#include<bits/stdc++.h>
const int maxn=200000+10;
int R,C,Q,size,dfsn,ch[maxn][27],fail[maxn],pos[maxn],dfsx[maxn],siz[maxn],len[maxn];
long long sum[5][maxn];
char s[600][600],S[maxn];
std::vector<int> e[maxn];
inline void dfs ( int u )
{
	dfsx[u]=++dfsn;siz[u]=1;
	for ( int v:e[u] ) dfs(v),siz[u]+=siz[v];
}
inline long long calc ( int u,int t ) { return sum[t][dfsx[u]+siz[u]-1]-sum[t][dfsx[u]-1]; }
signed main()
{
	scanf("%d%d%d",&R,&C,&Q);
	for ( int i=1;i<=R;i++ ) scanf(" %s",s[i]+1);
	for ( int i=1;i<=Q;i++ )
	{
		scanf(" %s",S+1);int u=0;
		for ( int j=1;S[j];j++ )
		{
			if ( !ch[u][S[j]^64] ) ch[u][S[j]^64]=++size;
			u=ch[u][S[j]^64];len[i]++;
		}
		pos[i]=u;
	}
	std::queue<int> q;
	for ( int i=1;i<=26;i++ ) if ( ch[0][i] ) q.push(ch[0][i]);
	while ( !q.empty() )
	{
		int u=q.front();q.pop();
		for ( int i=1;i<=26;i++ )
			if ( ch[u][i] ) fail[ch[u][i]]=ch[fail[u]][i],q.push(ch[u][i]);
			else ch[u][i]=ch[fail[u]][i];
	}
	for ( int i=1;i<=size;i++ ) e[fail[i]].push_back(i);
	dfs(0);
	for ( int i=1;i<=R;i++ ) for ( int j=1;j<=C;j++ )
	{
		int u=0,x=i,y=1;
		while ( y<j ) u=ch[u][s[x][y]^64],y++,sum[1][dfsx[u]]++;
		while ( x<=R ) u=ch[u][s[x][y]^64],x++,sum[1][dfsx[u]]++;
	}
	for ( int j=1;j<=C;j++ )
	{
		int u=0;
		for ( int i=1;i<=R;i++ ) u=ch[u][s[i][j]^64],sum[2][dfsx[u]]+=i,sum[3][dfsx[u]]++;
	}
	for ( int i=1;i<=R;i++ )
	{
		int u=0;
		for ( int j=1;j<=C;j++ ) u=ch[u][s[i][j]^64],sum[4][dfsx[u]]+=C-j;
	}
	for ( int t=1;t<=4;t++ ) for ( int i=1;i<=dfsn;i++ ) sum[t][i]+=sum[t][i-1];
	for ( int i=1;i<=Q;i++ ) printf("%lld\n",calc(pos[i],1)-calc(pos[i],2)+calc(pos[i],3)*len[i]-calc(pos[i],4));
	return 0;
}
